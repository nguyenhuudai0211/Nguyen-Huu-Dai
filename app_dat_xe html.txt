    <script>
        // Khởi tạo bản đồ
        var map = L.map('map').setView([10.7769, 106.7009], 13);
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: '© OpenStreetMap contributors'
        }).addTo(map);

        var currentRoute = null;
        var startMarker = null;
        var endMarker = null;
        var carMarker = null;
        var tripData = {}; // Lưu dữ liệu chuyến đi

        // Nút Tip Effect
        document.querySelectorAll('.tip-btn').forEach(btn => {
            btn.addEventListener('click', function() {
                document.querySelectorAll('.tip-btn').forEach(b => b.classList.remove('active'));
                this.classList.add('active');
            });
        });

        // Hàm tính toán và vẽ đường
        async function findRide() {
            var p_val = document.getElementById('pickup').value.split(',');
            var d_val = document.getElementById('dropoff').value.split(',');
            var p_name = document.getElementById('pickup').options[document.getElementById('pickup').selectedIndex].text;
            var d_name = document.getElementById('dropoff').options[document.getElementById('dropoff').selectedIndex].text;
            
            if (p_val[0] === d_val[0] && p_val[1] === d_val[1]) {
                alert("Trùng điểm đón và điểm đến!");
                return;
            }

            var lat1 = p_val[0], lon1 = p_val[1];
            var lat2 = d_val[0], lon2 = d_val[1];

            // Gọi OSRM API
            var url = `https://router.project-osrm.org/route/v1/driving/${lon1},${lat1};${lon2},${lat2}?overview=full&geometries=geojson`;
            
            try {
                var response = await fetch(url);
                var data = await response.json();
                var coords = data.routes[0].geometry.coordinates.map(c => [c[1], c[0]]); // Leaflet cần [lat, lon]
                var dist = data.routes[0].distance / 1000;

                // Xóa đường cũ
                if (currentRoute) map.removeLayer(currentRoute);
                if (startMarker) map.removeLayer(startMarker);
                if (endMarker) map.removeLayer(endMarker);
                if (carMarker) map.removeLayer(carMarker);

                // Vẽ đường mới
                currentRoute = L.polyline(coords, {color: '#00B14F', weight: 6}).addTo(map);
                startMarker = L.marker([lat1, lon1]).addTo(map);
                endMarker = L.marker([lat2, lon2]).addTo(map);
                map.fitBounds(currentRoute.getBounds());

                // Tính giá (Mô phỏng)
                var service = document.getElementById('service').value;
                var baseRate = service === "Xe Máy" ? 8000 : (service === "Ô Tô 4 Chỗ" ? 12000 : 16000);
                var basePrice = dist * baseRate;
                var surge = 1.3; // Mặc định nhân hệ số giá do trời nắng
                var total = Math.round(basePrice * surge);

                // Lưu dữ liệu để dùng sau
                tripData = { dist: dist, total: total, service: service, p_name: p_name, d_name: d_name, coords: coords };

                // Hiển thị Bill tạm tính
                document.getElementById('result-area').innerHTML = `
                    <div style='background:#fff;border:2px solid #00B14F;padding:20px;border-radius:12px;'>
                        <div style='display:flex;justify-content:space-between;border-bottom:1px solid #d1fae5;padding-bottom:10px;margin-bottom:10px;'>
                            <b style='color:#00B14F;'>📋 PHIẾU TÍNH TIỀN</b>
                            <span style='background:#f0fdf4;color:#00B14F;font-size:10px;padding:3px 8px;border-radius:10px;font-weight:bold;'>HỢP LỆ</span>
                        </div>
                        <div style='display:flex;justify-content:space-between;font-size:13px;margin-bottom:8px;'>
                            <span>Quãng đường:</span><b>${dist.toFixed(2)} km</b>
                        </div>
                        <div style='display:flex;justify-content:space-between;font-size:13px;margin-bottom:8px;'>
                            <span>Hệ số giá động:</span><b style='color:#00B14F;'>x${surge}</b>
                        </div>
                        <div style='border-top:1px dashed #e5e7eb;padding-top:15px;margin-top:10px;display:flex;justify-content:space-between;align-items:center;'>
                            <b style='font-size:16px;'>TỔNG TIỀN:</b>
                            <b style='color:#00B14F;font-size:24px;'>${total.toLocaleString('vi-VN')} đ</b>
                        </div>
                    </div>
                `;

                // Đổi nút
                document.getElementById('action-box').innerHTML = `
                    <div style="display:flex; justify-content:space-between; margin-top:15px;">
                        <button class="btn-cancel" onclick="resetUI()">HỦY</button>
                        <button class="btn-confirm" onclick="confirmRide()">XÁC NHẬN ĐẶT</button>
                    </div>
                `;
            } catch (e) {
                alert("Lỗi tải đường đi, vui lòng thử lại!");
            }
        }

        // Hủy
        function resetUI() {
            document.getElementById('result-area').innerHTML = "";
            document.getElementById('action-box').innerHTML = `<button class="btn-main" onclick="findRide()">TÌM CHUYẾN XE</button>`;
            if (currentRoute) map.removeLayer(currentRoute);
            if (startMarker) map.removeLayer(startMarker);
            if (endMarker) map.removeLayer(endMarker);
            if (carMarker) map.removeLayer(carMarker);
        }

        // Tạo UI Trạng thái tài xế
        function getDriverCard(state) {
            let c1 = state === 0 ? "#00B14F" : (state === 1 ? "#f59e0b" : "#3b82f6");
            let c2 = state === 0 ? "#f0fdf4" : (state === 1 ? "#fffbeb" : "#eff6ff");
            let txt1 = state === 0 ? "TÀI XẾ ĐANG ĐẾN" : (state === 1 ? "ĐANG TRONG CHUYẾN ĐI" : "ĐÃ KẾT THÚC");
            let txt2 = state === 0 ? "Đang tới" : (state === 1 ? "Di chuyển" : "Hoàn tất");
            let ic = tripData.service === "Xe Máy" ? "🏍️" : "🚗";

            return `
                <div style='background:${c2};border:2px solid ${c1};padding:15px;border-radius:12px;margin-top:15px;'>
                    <div style='display:flex;justify-content:space-between;align-items:center;border-bottom:1px solid ${c1};padding-bottom:10px;margin-bottom:10px;'>
                        <b style='font-size:16px;color:${c1};'>${txt1}</b>
                        <span style='background:${c1};color:white;padding:4px 10px;border-radius:12px;font-size:11px;font-weight:bold;'>${txt2}</span>
                    </div>
                    <div style='display:flex;align-items:center;'>
                        <div style='font-size:35px;margin-right:15px;'>${ic}</div>
                        <div>
                            <b style='font-size:15px;'>Nguyễn Hoàng K</b><br>
                            <small style='color:#6b7280;'>⭐ 4.9 | 59A-283.91</small>
                        </div>
                    </div>
                </div>
            `;
        }

        // Bấm Xác nhận -> Chạy Radar -> Xe chạy
        function confirmRide() {
            document.getElementById('booking-ui').style.display = 'none';
            document.getElementById('radar-ui').style.display = 'block';

            // Giả lập Radar 2 giây
            setTimeout(() => {
                document.getElementById('radar-ui').style.display = 'none';
                document.getElementById('booking-ui').style.display = 'block';
                document.getElementById('action-box').innerHTML = ""; // Ẩn nút
                
                // Hiện trạng thái 0 (Đang đến)
                document.getElementById('result-area').innerHTML += getDriverCard(0);

                // Cho xe chạy
                animateCar();
            }, 2000);
        }

        // Animation xe chạy bằng JavaScript mượt mà
        function animateCar() {
            var iconUrl = tripData.service === "Xe Máy" 
                ? "https://cdn-icons-png.flaticon.com/512/7191/7191059.png" 
                : "https://cdn-icons-png.flaticon.com/512/3085/3085330.png";
            
            var carIcon = L.icon({ iconUrl: iconUrl, iconSize: [40, 40] });
            carMarker = L.marker(tripData.coords[0], {icon: carIcon}).addTo(map);

            var i = 0;
            var coords = tripData.coords;
            var totalSteps = coords.length;
            var middleStep = Math.floor(totalSteps / 4); // Chạy đc 1/4 quãng đường thì đổi chữ

            var interval = setInterval(() => {
                if (i >= totalSteps) {
                    clearInterval(interval);
                    // Tới đích -> Trạng thái 2
                    document.getElementById('result-area').innerHTML = document.getElementById('result-area').innerHTML.replace(/<div style='background:.*<\/div>/s, getDriverCard(2));
                    
                    // Hiện Bill toàn màn hình sau 1.5 giây
                    setTimeout(showFullScreenBill, 1500);
                    return;
                }

                // Cập nhật vị trí
                carMarker.setLatLng(coords[i]);

                // Đổi trạng thái sang "Đang trong chuyến" khi đi được 1 chút
                if (i === middleStep) {
                    document.getElementById('result-area').innerHTML = document.getElementById('result-area').innerHTML.replace(/<div style='background:.*<\/div>/s, getDriverCard(1));
                }

                i++;
            }, 30); // Tốc độ cực nhanh (30ms mỗi tọa độ) để khỏi phải chờ
        }

        // Hiện Bill Tím toàn màn hình
        function showFullScreenBill() {
            var now = new Date();
            var timeStr = now.getHours() + ":" + now.getMinutes() + " " + now.getDate() + "/" + (now.getMonth()+1) + "/" + now.getFullYear();
            
            document.getElementById('bill-total').innerText = tripData.total.toLocaleString('vi-VN') + "đ";
            document.getElementById('bill-time').innerText = timeStr;
            document.getElementById('bill-pickup').innerText = tripData.p_name;
            document.getElementById('bill-dropoff').innerText = tripData.d_name;
            document.getElementById('bill-service').innerText = tripData.service;
            document.getElementById('bill-dist').innerText = tripData.dist.toFixed(1) + " km";

            document.getElementById('full-screen-bill').style.display = 'flex';
        }

        // Nút đặt lại từ đầu
        function resetApp() {
            document.getElementById('full-screen-bill').style.display = 'none';
            resetUI();
        }
